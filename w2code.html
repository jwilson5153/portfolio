                function add() {
                    const input = document.getElementById("numbers").value;

                function stepByNumber() {
                    var input = document.getElementById("numbers").value;
                    input = parseInt(input);
                    if (typeof input == "number") {
                        let total = 0;
                        for (let i = 0; i <= input; i++) {
                            total += i;
                        }
                        document.getElementById("results").innerHTML = total;
                    }
                    else {
                        document.getElementById("results").innerHTML = "Not a Number!";
                    }
                };

                function addAgain() {
                    var number1 = document.getElementById("numbers").value;
                    var number2 = document.getElementById("numberTwo").value;
                    number1 = parseInt(number1);
                    number2 = parseInt(number2);
                    var answer = (number1 + number2);
                    document.getElementById("results").innerHTML = answer;
                };

                </script>
